# Type Reference

All public types exported by Run-IQ packages.

## Core Types (`@run-iq/core`)

### Rule

```typescript
interface Rule {
  readonly id: string;
  readonly version: number;
  readonly model: string;
  readonly params: unknown;
  readonly condition?: Expression;
  readonly priority: number;
  readonly effectiveFrom: Date;
  readonly effectiveUntil: Date | null;
  readonly tags: readonly string[];
  readonly checksum: string;
  readonly schemaVersion?: string;
}
```

### Expression

```typescript
interface Expression {
  readonly dsl: string;
  readonly value: unknown;
}
```

### EvaluationInput

```typescript
interface EvaluationInput {
  readonly data: Record<string, unknown>;
  readonly requestId: string;
  readonly meta: {
    readonly tenantId: string;
    readonly userId?: string;
    readonly tags?: readonly string[];
    readonly context?: Record<string, unknown>;
    readonly effectiveDate?: Date;
  };
}
```

### EvaluationResult

```typescript
interface EvaluationResult {
  readonly requestId: string;
  readonly value: unknown;
  readonly breakdown: readonly BreakdownItem[];
  readonly appliedRules: readonly Rule[];

  readonly trace: EvaluationTrace;
  readonly snapshotId: string;
  readonly engineVersion: string;
  readonly pluginVersions: Record<string, string>;
  readonly dslVersions: Record<string, string>;
  readonly timestamp: Date;
  readonly meta?: Record<string, unknown>;
}
```

### DSLEvaluator

```typescript
interface DSLEvaluator {
  readonly dsl: string;

```

### PPEPlugin

```typescript
interface PPEPlugin {
  readonly name: string;
  readonly version: string;
  readonly models?: CalculationModel[];

  onInit?(context: PluginContext): void;
  beforeEvaluate?(input: EvaluationInput, rules: ReadonlyArray<Rule>): BeforeEvaluateResult;
  afterEvaluate?(input: EvaluationInput, result: EvaluationResult): EvaluationResult;
  onError?(error: PPEError, input: EvaluationInput): void;
  teardown?(): void;
}
```

### BeforeEvaluateResult

Returned by a plugin's `beforeEvaluate` hook, allowing it to modify both the input data and the rules array before core execution.

```typescript
interface BeforeEvaluateResult {
  /** The potentially modified input */
  readonly input: EvaluationInput;
  /** The potentially modified rules array */
  readonly rules: ReadonlyArray<Rule>;
}
```


### Snapshot

```typescript
interface Snapshot {

  readonly pluginVersions: Record<string, string>;

  readonly inputSnapshot: EvaluationInput;

```

### PPEEngineConfig

```typescript
interface PPEEngineConfig {
  readonly plugins: PPEPlugin[];
  readonly dsls: DSLEvaluator[];
  readonly snapshot?: ISnapshotAdapter;
  readonly strict?: boolean;
  readonly timeout?: {
    readonly dsl?: number;
    readonly hook?: number;
    readonly pipeline?: number;
  };
  readonly onConflict?: 'throw' | 'first';
  readonly onChecksumMismatch?: 'throw' | 'skip';
  readonly dryRun?: boolean;
}
```
